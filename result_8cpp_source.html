40 fp::Result<double> safe_sqrt(double x) {
41  if (x < 0) {
42  return tl::make_unexpected(fp::InvalidArgument(
43  fmt::format("sqrt of value < 0.0 is undefined: {}", x)));

49 fp::Result<double> do_math(double x) {
50  return fp::make_result(x).and_then(divide_4_by).and_then(safe_sqrt);
